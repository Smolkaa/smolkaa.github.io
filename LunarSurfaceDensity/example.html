Example

using Distributed   # enable parallelized computing
addprocs(4)         # enable 4 workers to split the Monte-Carlo simulation onto

using DelimitedFiles    # for writing of results to disk

@everywhere include(joinpath("..","src","ExosphereSimulations.jl")) # load local package file
@everywhere using .ExosphereSimulations                             # enable ExosphereSimulations

# defines a grid with 32400 equally sized elements on half a sphere with radis `MOON_RADIUS`
grid = FSpiralHalfSGrid(MOON_RADIUS, 32400)

# defines the physical input parameters for the helium simulation
physics = LSDPhysics(
    4.0026,                 # atomic mass unit
    1.5e6,                  # global source rate [cm-2 s-1]
    8.0e6,                  # Sun-side source rate [cm-2 s-1]
    5.25e-8,                # photo-destruction rate [s-1]
    0.0,                    # heat of adsorption [J mol-1]
    0.0,                    # ion recycling fraction [-]
    :MaxwellBoltzmannFlux   # velocity distribution
)

# defines the solver options
opt = LSDOptions(;  # uses default values
    N=1000,         # adjusts the number Monte-Carlo steps
)

# defines the problem struct to pass to the solver function
prob = LSDProblem(grid, physics, opt)

# solve the problem
sol = solve(prob)

# save surface number densities n0 on the numerical grid, and on a structured grid.
savesol(joinpath(@__DIR__, "csv", "He_n0"), sol)
gridsavesol(joinpath(@__DIR__, "csv", "He_n0grid"), sol, collect(range(-pi, pi, length=362))[2:end-1], collect(range(0, pi/2, length=182))[2:end-1])
